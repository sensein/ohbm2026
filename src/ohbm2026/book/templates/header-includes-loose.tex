% LaTeX preamble for the plain (book class) PDF rendering — Stage 12
% LOOSE margin preset. Loaded by pandoc via `-H header-includes-loose.tex`
% when the operator passes `ohbmcli book --margins=loose`.
%
% Recovers the LaTeX `book` class default margins (~1 in all sides).
% Use case: reproducing the pre-Stage-12 layout for archival or
% comparison purposes. No `\usepackage{geometry}` import — the book
% class default applies as-is.

\usepackage{makeidx}
\makeindex
\usepackage{graphicx}
% Pandoc's --resource-path injects the book-output directory so
% relative `fig_assets/...` paths resolve.
\usepackage{microtype}

% Stage 12 US3 — `longtable` environment for the new 3-column TOC
% (Poster | Title | Page). Multi-page table; standard LaTeX extra
% package that ships with Tectonic + every TeX Live distribution.
\usepackage{longtable}

% xelatex's default math font is Latin Modern Math (built into every
% TeX distribution incl. Tectonic), which has the full Greek alphabet
% plus standard scientific operators. The html_to_md preprocessor
% wraps Greek letters + math operators in `\(...\)` so they render
% via this default math font without needing fontspec / system fonts.

\usepackage{hyperref}
\hypersetup{
  colorlinks=true,
  linkcolor=black,
  urlcolor=blue,
}
